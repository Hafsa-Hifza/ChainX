/* ══ Connect Buttons ══ */
function toggleConnect(btn) {
    var row = btn.closest('.sp-item') || btn.closest('.connect-item');
    if (btn.classList.contains('done')) {
        btn.classList.remove('done');
        btn.textContent = 'Follow';
        if (row) row.classList.remove('followed');
    } else {
        btn.style.transform = 'scale(0.92)';
        btn.style.opacity = '0.6';
        setTimeout(function(){
            btn.classList.add('done');
            btn.textContent = 'Following';
            btn.style.transform = '';
            btn.style.opacity = '';
            if (row) row.classList.add('followed');
        }, 150);
    }
}

/* ══ Follow Back Toggle ══ */
var followBackManager = {
    pauseTimer: null,
    PAUSE_DELAY: 2000,
    
    toggle: function(btn) {
        var row = btn.closest('.sp-item') || btn.closest('.connect-item');
        var isInFullList = btn.closest('#followBackList') !== null;
        
        if (btn.classList.contains('friends')) {
            btn.classList.remove('friends');
            btn.classList.add('fb');
            btn.textContent = 'Follow Back';
            row.classList.remove('completed');
            row.style.maxHeight = '';
            return;
        }
        
        btn.classList.remove('fb');
        btn.classList.add('friends');
        btn.textContent = 'Friends ✓';
        row.classList.add('completed');
        
        if (isInFullList) {
            row.style.maxHeight = row.offsetHeight + 'px';
            this.scheduleBatchExit();
        }
    },
    
    scheduleBatchExit: function() {
        var self = this;
        if (this.pauseTimer) clearTimeout(this.pauseTimer);
        this.pauseTimer = setTimeout(function() {
            self.exitCompletedItems();
        }, this.PAUSE_DELAY);
    },
    
    exitCompletedItems: function() {
        var completedItems = document.querySelectorAll('#followBackList .sp-item.completed:not(.exiting)');
        if (completedItems.length === 0) return;
        
        completedItems.forEach(function(item, index) {
            setTimeout(function() {
                item.classList.add('exiting');
                setTimeout(function() {
                    item.remove();
                    updateFollowBackBadge();
                    checkEmptyState();
                }, 400);
            }, index * 80);
        });
    }
};

function followBack(btn) {
    followBackManager.toggle(btn);
}

function checkEmptyState() {
    var list = document.getElementById('followBackList');
    var empty = document.getElementById('followBackEmpty');
    if (list && list.children.length === 0 && empty) {
        empty.style.display = '';
    }
}

function updateFollowBackBadge() {
    var list = document.getElementById('followBackList');
    var countEl = document.getElementById('followedYouCount');
    if (list) {
        var count = list.querySelectorAll('.sp-item:not(.exiting)').length;
        if (countEl) countEl.textContent = count;
    }
}

/* ══ Unfriend Popup & Friends Management ══ */
var unfriendTarget = null;

function showUnfriendPopup(btn) {
    var row = btn.closest('.friend-item');
    if (!row) return;
    unfriendTarget = row;
    var name = row.getAttribute('data-name') || row.querySelector('.sp-name').textContent;
    var avatarSrc = row.querySelector('img').src;
    
    document.getElementById('ufName').textContent = name;
    var avatarEl = document.getElementById('ufAvatar');
    avatarEl.innerHTML = '<img src="' + avatarSrc + '" alt="">';
    
    var overlay = document.getElementById('unfriendOverlay');
    overlay.offsetHeight;
    requestAnimationFrame(function() {
        overlay.classList.add('show');
    });
}

function closeUnfriendPopup() {
    var overlay = document.getElementById('unfriendOverlay');
    overlay.classList.remove('show');
    unfriendTarget = null;
}

function confirmUnfriend() {
    if (!unfriendTarget) return;
    var row = unfriendTarget;
    closeUnfriendPopup();
    row.style.maxHeight = row.offsetHeight + 'px';
    row.offsetHeight;
    
    requestAnimationFrame(function() {
        row.classList.add('removing');
        setTimeout(function() {
            row.remove();
            updateFriendsCount();
        }, 450);
    });
}

function updateFriendsCount() {
    var list = document.getElementById('friendsList');
    var count = list ? list.querySelectorAll('.friend-item:not(.removing)').length : 0;
    var headerCount = document.getElementById('friendsHeaderCount');
    var pillBadge = document.getElementById('friendsPillBadge');
    
    if (headerCount) headerCount.textContent = count + (count === 1 ? ' friend' : ' friends');
    if (pillBadge) {
        pillBadge.textContent = count;
        if (count === 0) pillBadge.style.display = 'none';
    }
    
    var emptyEl = document.getElementById('friendsEmpty');
    if (count === 0 && emptyEl) {
        emptyEl.style.display = '';
    }
}

/* ══ Pull to Refresh ══ */
function initPullToRefresh(){
    document.querySelectorAll('[data-ptr]').forEach(function(el){ new PTR(el) });
}
function PTR(el){
    this.el=el;this.ind=el.querySelector('.ptr-indicator');
    this.svg=el.querySelector('.ptr-svg');this.path=el.querySelector('.ptr-path');
    if(!this.ind||!this.path) return;
    this.TH=55;this.MX=85;this.AS=35;
    this.startY=0;this.d=0;this.busy=false;this.pull=false;
    this.resetArc();
    var me=this;
    el.addEventListener('touchstart',function(e){me.ts(e)},{passive:true});
    el.addEventListener('touchmove',function(e){me.tm(e)},{passive:false});
    el.addEventListener('touchend',function(){me.te()},{passive:true});
}
PTR.prototype.resetArc=function(){
    this.path.setAttribute('d','');
    this.path.style.opacity='0';
    this.svg.style.transform='';
};
PTR.prototype.ts=function(e){
    if(this.busy||this.el.scrollTop>0) return;
    this.startY=e.touches[0].clientY;
    this.pull=true;this.d=0;
    this.ind.classList.remove('snapping','releasing','refreshing');
    this.ind.style.transition='none';
    this.el.style.transition='none';
};
PTR.prototype.mkPath=function(p){
    if(p<.01)return'';
    var cx=50,cy=50,rI=17.5,rO=24.5,rM=21,sa=38,span=p*290,ea=sa+span;
    var la=span>180?1:0;
    function pt(r,d){var a=d*Math.PI/180;return[cx+r*Math.cos(a),cy+r*Math.sin(a)]}
    var tI=pt(rI,sa),tO=pt(rO,sa),bI=pt(rI,ea),bO=pt(rO,ea);
    if(p<=.65)return'M'+tI[0]+' '+tI[1]+'L'+tO[0]+' '+tO[1]+'A '+rO+' '+rO+' 0 '+la+' 1 '+bO[0]+' '+bO[1]+'L'+bI[0]+' '+bI[1]+'A '+rI+' '+rI+' 0 '+la+' 0 '+tI[0]+' '+tI[1]+'Z';
    var hp=Math.min((p-.65)/.35,1),hRI=rI-hp*(rI-10),hRO=rO+hp*(32-rO),hL=hp*13;
    var hbI=pt(hRI,ea),hbO=pt(hRO,ea),bM=pt(rM,ea);
    var ta=(ea+90)*Math.PI/180,tx=bM[0]+hL*Math.cos(ta),ty=bM[1]+hL*Math.sin(ta);
    return'M'+tI[0]+' '+tI[1]+'L'+tO[0]+' '+tO[1]+'A '+rO+' '+rO+' 0 '+la+' 1 '+bO[0]+' '+bO[1]+'L'+hbO[0]+' '+hbO[1]+'L'+tx+' '+ty+'L'+hbI[0]+' '+hbI[1]+'L'+bI[0]+' '+bI[1]+'A '+rI+' '+rI+' 0 '+la+' 0 '+tI[0]+' '+tI[1]+'Z';
};
PTR.prototype.tm=function(e){
    if(!this.pull||this.busy)return;
    var raw=e.touches[0].clientY-this.startY;
    if(raw<0||this.el.scrollTop>0){this.d=0;return}
    e.preventDefault();
    this.d=Math.min(raw*.4,this.MX);
    this.ind.style.transform='translateX(-50%) translateY('+this.d+'px)';
    
    if(this.d<8)this.ind.style.opacity=0;
    else if(this.d<35)this.ind.style.opacity=(this.d-8)/27;
    else this.ind.style.opacity=1;
    
    if(this.d<this.AS){
        this.resetArc();
        this.el.style.transform='translateY('+(this.d*.15)+'px)';return;
    }
    var p=Math.min((this.d-this.AS)/(this.TH-this.AS),1);
    this.path.setAttribute('d',this.mkPath(p));
    this.path.style.opacity=(.2+p*.8);
    
    var rot;
    if(this.d<=this.TH){rot=-120+p*120}
    else{var op=Math.min((this.d-this.TH)/(this.MX-this.TH),1);rot=op*90}
    this.svg.style.transform='rotate('+rot+'deg)';
    this.el.style.transform='translateY('+(this.d*.15)+'px)';
};
PTR.prototype.te=function(){
    if(!this.pull)return;this.pull=false;
    this.d>=this.TH?this.go():this.no();
};
PTR.prototype.go=function(){
    this.busy=true;
    this.path.setAttribute('d',this.mkPath(1));
    this.path.style.opacity='1';
    this.ind.style.transition='';
    this.ind.classList.add('snapping');
    this.ind.style.transform='translateX(-50%) translateY(38px)';
    this.ind.style.opacity='1';
    this.el.style.transition='';
    this.el.classList.add('shifting');
    this.el.style.transform='translateY(12px)';
    var me=this;
    setTimeout(function(){
        me.svg.style.transform='';
        me.ind.classList.add('refreshing');
    },230);
    setTimeout(function(){me.done()},1500);
};
PTR.prototype.done=function(){
    var me=this;
    me.ind.classList.remove('refreshing','snapping');
    me.ind.classList.add('releasing');
    me.ind.style.transform='translateX(-50%) translateY(-38px)';
    me.ind.style.opacity='0';
    me.el.classList.remove('shifting');
    me.el.classList.add('releasing');
    me.el.style.transform='';
    setTimeout(function(){
        me.ind.classList.remove('releasing');
        me.ind.style.transition='';
        me.el.classList.remove('releasing');
        me.el.style.transition='';
        me.resetArc();me.busy=false;
    },380);
};
PTR.prototype.no=function(){
    var me=this;
    me.ind.classList.add('releasing');
    me.ind.style.transform='translateX(-50%) translateY(-38px)';
    me.ind.style.opacity='0';
    me.el.classList.add('releasing');
    me.el.style.transform='';
    setTimeout(function(){
        me.ind.classList.remove('releasing');
        me.ind.style.transition='';
        me.el.classList.remove('releasing');
        me.el.style.transition='';
        me.resetArc();
    },380);
};

document.addEventListener('DOMContentLoaded', function() { 
    initPullToRefresh();
